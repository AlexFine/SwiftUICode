Activity Indicator

Quickly add a custom activity indicator to your app. 


import SwiftUI
import ActivityIndicator

struct ContentView: View {
  @State var isLoading = false

  var body: some View {
    ActivityIndicator(loading: self.$isLoading)
  }
}

TextView

Add a multiline text field into your app. 


import SwiftUI
import TextView

struct ContentView: View {
  @State var text = ""
  @State var isActive = false

  var body: some View {
    TextView(
      text: $text,
      isEditing: $isActive
    )
  }
}
