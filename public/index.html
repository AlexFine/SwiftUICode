Meet iSwiftUI

A library of 9 SwiftUI packages.

iActivityIndicator

Quickly implement & easily customize a great looking activity indicator.

 
import SwiftUI
import iActivityIndicator

struct ContentView: View {
  var body: some View {
      iActivityIndicator()
  }
}

iPages

Quickly implement swipable views.

 
import SwiftUI
import iPages
import iGraphics

struct ContentView: View {
    @State var currentPage: Int = 0

    var body: some View {
        iPages([iGraphics(), iGraphics()], currentPage: $currentPage)
    }
}

iLineChart

Quickly implement & easily customize great looking line charts. 

import SwiftUI
import LineChart

struct ContentView: View {
  var body: some View {
      LineChart(data: [8,32,11,23,40,28,15,20,30,25])
  }
}

iColors

Quickly add modern colors to your app.

 
import SwiftUI
import iColors

struct ContentView: View {
    var body: some View {
        Rectangle().fill(Colors.NeonBlue)
    }
}

iPhoneNumberTextField

Quickly implement a textfield designed for phone number.

 
import SwiftUI
import iPhoneNumberTextField

struct ContentView: View {
    @State var text: String = ""
    @State var isEditing: Bool = false

    var body: some View {
        iPhoneNumberTextField("Placeholder", text: $text, isEditing: $isEditing)
            .style()
    }
}

iGraphics

import SwiftUI

struct ContentView: View {

    }
}

iTextField

Quickly implement a textfield with all the UIKit capabilities in SwiftUI.

 
import SwiftUI
import iTextField

struct ContentView: View {
    @State var text: String = ""
    @State var isEditing: Bool = false

    var body: some View {
        iTextField("Placeholder", text: $text, isEditing: $isEditing)
    }
}

iPaymentButton

Quickly implement & easily customize Apple's required payment button.

 
import SwiftUI
import iPaymentButton

struct ContentView: View {
    var body: some View {
        iPaymentButton(action: {
            // Add your custom payment code here 
        })
    }
}
