SwiftUILibrary

Colors

Quickly add modern colors to your app.

1.

Install Colors.

2.

Add Colors to your project. 

import SwiftUI
import Colors

struct ContentView: View {
  var body: some View {
      Rectangle().fill(Colors.NeonBlue)
  }
}

3.

Customize your colors.


Examples

Ocean

Use these colors to convey health, trust, & security. 

import SwiftUI
import Colors

struct ContentView: View {
    var body: some View {
        VStack{
          HStack{
            Rectangle().fill(Colors.LightGreen).cornerRadius(5)
            Rectangle().fill(Colors.NeonGreen).cornerRadius(5)
            Rectangle().fill(Colors.DarkGreen).cornerRadius(5)
          }
          HStack{
            Rectangle().fill(Colors.LightOceanBlue).cornerRadius(5)
            Rectangle().fill(Colors.NeonOceanBlue).cornerRadius(5)
            Rectangle().fill(Colors.DarkOceanBlue).cornerRadius(5)
          }
          HStack{
            Rectangle().fill(Colors.LightBlue).cornerRadius(5)
            Rectangle().fill(Colors.NeonBlue).cornerRadius(5)
            Rectangle().fill(Colors.DarkBlue).cornerRadius(5)
          }
        }
    }
}
